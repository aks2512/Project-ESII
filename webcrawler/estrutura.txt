    function listar() {
        $('#load').show("fast");
        $.ajax({
            type: "get",
            dataType: "json",
            url: "http://www.licitacao.pmmc.com.br/Transparencia/vencimentos2",
            success: function (data) {
                var html = '';
                var tabela = $("#tabela");
                tabela.html('');
                $.each(data.servidores, function (id, valor) {
                    html = '<tr><td>' + (id + 1) + '</td><td>' + valor.nome + '</td><td>' + valor.cargo + '</td><td>' + valor.rendimentos + '</td><td><a href="' + valor.rgf + '"> <span class="glyphicon glyphicon-plus" aria-hidden="true"></span></a></td></tr>';
                    tabela.append(html);
                });

                $('#tbFuncionarios').DataTable({
                    "ordering": false,
                    "oLanguage": {
                        "sProcessing": "Processando...",
                        "sLengthMenu": "Mostrar _MENU_ registros",
                        "sZeroRecords": "Não foram encontrados resultados",
                        "sInfo": "_START_ até _END_ de _TOTAL_ registros",
                        "sInfoEmpty": "0 até 0 de 0 registros",
                        "sInfoFiltered": "",
                        "sInfoPostFix": "",
                        "sSearch": "Buscar:",
                        "sUrl": "",
                        "oPaginate": {
                            "sFirst": "<<",
                            "sPrevious": "<",
                            "sNext": ">",
                            "sLast": ">>"
                        }
                    }
                });

            },
            error: function (dados) {
                alert('Erro');
            },
            complete: function () {
                $('#load').hide("fast");
            }
        });
    }

    $(document).ready(function () {
        listar();
        $('[data-toggle="tooltip"]').tooltip();

        $('#tabela').on('click', 'a', function (e) {
            $('#load').show("fast");
            var rgf = $(this).attr('href');
            $.ajax({
                type: "get",
                dataType: "json",
                url: "http://www.licitacao.pmmc.com.br/Transparencia/detalhamento?rgf=" + rgf,
                success: function (data) {
                    $('#nome').text(data.nome);
                    $('#cargo').text(data.cargo);
                    $('#referencia').text(data.referencia);
                    $('#regime').text(data.regime);

                    $('#bruto').text('');
                    $('#liquido').text('');
                    $('#descontos').text('');

                    $.each(data.totais, function (id, valor) {
                        if (valor.nome.trim() === "TOTAL VENCIM/TO") {
                            $('#bruto').text(valor.valor);
                        }
                        if (valor.nome.trim() === "TOTAL LIQUIDO") {
                            $('#liquido').text(valor.valor);
                        }
                        if (valor.nome.trim() === "TOTAL DESCONTOS") {
                            $('#descontos').text(valor.valor);
                        }
                    });


                    //remuneracao
                    var html = '';
                    var tabela = $("#tbRemuneracao");
                    tabela.html('');
                    $.each(data.rendimentos, function (id, valor) {
                        html = '<tr><td>' + valor.nome + '</td><td>' + valor.valor + '</td></tr>';
                        tabela.append(html);
                    });

                    //remuneracao
                    var html = '';
                    var tabela = $("#tbDescontos");
                    tabela.html('');
                    $.each(data.descontos, function (id, valor) {
                        html = '<tr><td>' + valor.nome + '</td><td>' + valor.valor + '</td></tr>';
                        tabela.append(html);
                    });
                    //Outros
                    $('#outros').text(data.outros[0].valor);
                    //Abrir Janela
                    $('#modalFuncionario').modal('show');

                },
                error: function (dados) {
                    alert('Erro');
                },
                complete: function () {
                    $('#load').hide("fast");
                }
            });
            e.preventDefault();
            return false;
        });

        $('#dados').on('click', '.close', function () {
            $(this).parent().parent().hide('fast', function () {
                $(".table-striped").show('fast');
            });

            return false;
        });

        $('#tabela').on('click', '.mais', function () {
            var id = $(this).attr('data');
            $(".table-striped").hide('fast', function () {
                $("#d-" + id).show('fast');
            });


            return false;
        });

        




    });


</script>
